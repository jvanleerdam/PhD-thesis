\section{Decay-Rate Equations}
\label{sec:pheno_equations}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% approximation of differential decay rate, CPV observables in decay rate,
% phi_s and C_s parameters, |lambda_s| used for historical reasons,
% correlations between Gamma, DeltaGamma, |A_0|^2, and |A_perp|^2,
% correlation between Deltam and deltaperp (sin(d_perp) * cos(Dm*t) + sin(Dm*t) in para/perp term),
% correlation between delta_para and delta_perp
% ambiguities in phases

\begin{equation}
  \begin{aligned}
    \label{eq:timeCPEvenOdd}
    \timeAmp[i,+1]^*\timeAmp[i,+1]^{\phantom{*}} + \timeAmp[i,-1]^*\timeAmp[i,-1]^{\phantom{*}}
      &\propto \eGst\left[ \cDGs - \eta_i\,\sDGs \right]  \\
      &= \left\{\begin{array}{ll}
                  e^{-\GL\,t}  &  \text{if } \eta_i = +1 \text{ (CP even)}  \\
                  e^{-\GH\,t}  &  \text{if } \eta_i = -1 \text{ (CP odd)}
         \end{array}\right.
  \end{aligned}
\end{equation}

\begin{equation}
  \label{eq:timeOscill}
  \timeAmp[i,+1]^*\timeAmp[i,+1]^{\phantom{*}} - \timeAmp[i,-1]^*\timeAmp[i,-1]^{\phantom{*}}
    \propto \eGst\left[ \Csi\,\cDms + \eta_i\,\phisi\,\sDms \right]
\end{equation}
